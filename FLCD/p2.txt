def int main():
	int n
	read(n)
	if [prime(n)]:
		print(n, ' is prime')
	else:
		print(n, ' is not prime')
	end

def bool prime(int n):
	if [n < 2]:
		ret false
	elif [n =: 2]:
		ret true
	elif [n % 2 =: 0]:
		ret false
	else:
		for [int d = 3, d * d <= n, d += 2]:
			if [n % d =: 0]:
				ret false
	end
	ret true 
end
